### DYNAMIC STRING ###
# local_string array contains the name of the array
# the actual string is stored in; think C pointers

# generally not recommended because of major use of meta_trigger

defined_text = {
	name = GetDynamicLocalString
	
	text = {
		trigger = {
			set_temp_variable = { n = 0 }
			always = yes
		}
		localization_key = "[GetStringDynamicInner]"
	}
}

	name = GetStringDynamicInner
	
	text = {
		trigger = {
			meta_trigger = {
				text = {

			}
		}
		localization_key = ""
