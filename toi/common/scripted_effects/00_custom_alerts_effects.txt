###GENERIC EFFECTS###
custom_alerts_add_alert = {
	hidden_effect = {
		ROOT = {
			if = {
				limit = {
					is_ai = no
					NOT = {
						is_in_array = {
							array = local_alerts_array
							value = custom_alerts_add_target
						}
						is_in_array = {
							array = local_alerts_disabled_array
							value = custom_alerts_add_target
						}
					}
				}
				add_to_array = {
					array = local_alerts_array
					value = custom_alerts_add_target
				}
				meta_effect = {
					text = {
						scoped_sound_effect = "custom_alerts_sound_[ID]"
					}
					ID = "[?custom_alerts_add_target]"
				}
			}
		}
	}
}

		ROOT = {

				}
				remove_from_array = {
					array = local_alerts_array
					value = custom_alerts_remove_target
				}
			}

					is_ai = no

					}
				}
				remove_from_array = {
					array = local_alerts_disabled_array
					value = custom_alerts_remove_target
				}
			}
		}
	}
}
