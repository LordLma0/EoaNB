	id = denmark_economy.11
	title = {
		trigger = { NOT = { has_country_flag = DEN_sweden_denies_all_wood_trade_flag } }
		text = denmark_economy.11.t_denied
	}
	title = {
		trigger = { has_country_flag = DEN_sweden_denies_all_wood_trade_flag }
		text = denmark_economy.11.t_denied_categorically
	}
	desc = {
		trigger = { NOT = { has_country_flag = DEN_sweden_denies_all_wood_trade_flag } }
		text = denmark_economy.11.d_denied
	}
	desc = {
		trigger = { has_country_flag = DEN_sweden_denies_all_wood_trade_flag }
		text = denmark_economy.11.d_denied_categorically
	}
		
	is_triggered_only = yes
	
	option = {
		name = denmark_economy.11.a
		ai_chance = { base = 100 }
		if = {
			limit = {
				NOT = { has_country_flag = DEN_sweden_denies_all_wood_trade_flag }
			}
			add_opinion_modifier = { target = SWE modifier = DEN_SWE_WOOD_TRADE_DENIED_REL }
			reverse_add_opinion_modifier = { target = SWE modifier = DEN_SWE_WOOD_TRADE_DENIED_REL }
		}
		else = {
			add_opinion_modifier = { target = SWE modifier = DEN_SWE_WOOD_TRADE_DENIED_CATEGORICALLY_REL }
			reverse_add_opinion_modifier = { target = SWE modifier = DEN_SWE_WOOD_TRADE_DENIED_CATEGORICALLY_REL }
		}
		hidden_effect = {
			clr_country_flag = DEN_asking_for_swedish_wood_flag
		}
	}
}

#SWE - Denmark Asks to Import our Iron
country_event = {
	id = denmark_economy.12
	title = denmark_economy.12.t
	desc = denmark_economy.12.d
		
	is_triggered_only = yes
	
	option = {
		name = denmark_economy.12.a
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				NOT = {
					any_owned_state = {
						has_resources_amount = {
							resource = iron
							amount > 6
						}
					}
				}
			}
			modifier = {
				factor = 0.5
				NOT = {
					any_owned_state = {
						has_resources_amount = {
							resource = iron
							amount > 12
						}
					}
				}
			}
		}
		custom_effect_tooltip = DEN_SWE_trade_accepted_effect_tt
		DEN = { country_event = { id = denmark_economy.13 } }
	}
	
	option = {
		name = denmark_economy.12.b
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				NOT = {
					any_owned_state = {
						has_resources_amount = {
							resource = iron
							amount > 6
						}
					}
				}
			}
			modifier = {
				add = 50
				NOT = {
					any_owned_state = {
						has_resources_amount = {
							resource = iron
							amount > 12
						}
					}
				}
			}
		}
		custom_effect_tooltip = DEN_SWE_trade_denied_effect_tt
		DEN = { country_event = { id = denmark_economy.14 } }
	}
	
	option = {
		name = denmark_economy.12.c
		ai_chance = { base = 0 }
		hidden_effect = {
			DEN = {
				set_country_flag = DEN_sweden_denies_all_iron_trade_flag
			}
		}
		custom_effect_tooltip = DEN_SWE_iron_trade_denied_forever_effect_tt
		DEN = { country_event = { id = denmark_economy.14 } }
	}
}
